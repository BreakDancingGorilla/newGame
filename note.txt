  class ACTOR {
    constructor(x,y,width,height){
      this.y = x;
      this.x = y;
      this.width = width;
      this.height = height
      this.screenX = 0;
      this.screenY = 0;
      this.addSelfToList();
    }
    draw(){
      this.screenX = this.x - camera.x;
      this.screenY = this.y - camera.y;
      ctx.fillRect(this.screenX,this.screenY,this.width,this.height);
    }
    addSelfToList(){
      actorList[actorList.length] = this;
    }
  }

      class NPC extends OBJECTLIST {
      constructor(speed, behavior){
        this.speed = speed;
        this.behavior = behavior; ///PASSIVE, NEUTRAL, AGGRESSIVE
        this.hunger = 100;
        this.stamina = 100;
        this.state = "roaming"; ///Attacking, roaming, eating, fleeing, 
      }
      update(){
        if (this.state = "roaming") {
          randomPointNearObj(this,500);
        }
      }
